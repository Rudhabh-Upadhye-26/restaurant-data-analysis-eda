# 4. Filling Missing Values & Preparing the Data

# 1. Fixingg missing values in 'establishment' (categorical)
df['establishment'] = df['establishment'].fillna(df['establishment'].mode()[0])
# establishment tells what type of restaurant it is.


# 2. Fixing missing values in 'address column'(categorical)
df['address'] = df['address'].fillna(df['address'].mode()[0])
# Using the most common address makes sense --> replacing blank with typical value.


# 3. Fixing missing values in 'zipcode'  --> ( numerical/categorical (depends on data type) )       
if df['zipcode'].dtype != 'object':                                   # If zipcode is numeric --> fill with median
    df['zipcode'] = df['zipcode'].fillna(df['zipcode'].median())
else:                                                                  # If zipcode is stored as object/strings --> fill with mode
    df['zipcode'] = df['zipcode'].fillna(df['zipcode'].mode()[0])


# 4. Fixingg missing values in 'cuisines'
df['cuisines'] = df['cuisines'].fillna(df['cuisines'].mode()[0])
# Mode makes sense because missing cuisine --> use the common cuisine.


# 5. Fixing missing values in 'timings' --> (categorical text)
df['timings'] = df['timings'].fillna('Not Available')
# Missing timing = restaurant didn’t share --> “Not Available” is perfect.


# 6. Fixing missing values in 'highlights' --> (categorical text)
# Highlights = special features or services that the restaurant offers.
df['highlights'] = df['highlights'].fillna("No info given")
# Because highlights tell special features, not every restaurant has the same features.


# 7. Fixingg missing values in 'opentable_support' --> (numerical (0/1 support))
df['opentable_support'] = df['opentable_support'].fillna(0)
# Values are 0 or 1 --> if missing, assume 0 (not supported).
